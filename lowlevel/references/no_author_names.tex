
\guideline[g:references:no_author_names]
    {Avoid mentioning authors by name.}

\goodbadexample[{\cite[Sec.~I.A.]{Wetzlinger2023TAC}}]{
    \highlightpart{Gan et al.} have shown that the exact reachable set can only be computed in rare special cases [7].
}{
    The exact reachable set can only be computed in rare special cases \highlightpart{[7]}.
}

\noindent Mentioning some (but not all) authors by name raises questions regarding the importance of their work---or their persona---with respect to other cited work.
It is better to discuss related work solely at the content level to avoid favoritism or the impression of personal attacks.
