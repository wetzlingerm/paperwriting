
\guideline[g:preliminaries:merge]
    {Merge short problem statements to the end of the preliminaries.}

    \textit{III. Problem Statement} \\
    As the exact reachable set $\mathcal{R}(t)$ cannot be computed for general nonlinear systems [2], our goal is to compute a tight inner approximation $\widecheck{\mathcal{R}}(t) \subseteq \mathcal{R}(t)$ instead.
}{
    \location{End of Preliminaries}
    As the exact reachable set $\mathcal{R}(t)$ cannot be computed for general nonlinear systems [2], our goal is to compute a tight inner approximation $\widecheck{\mathcal{R}}(t) \subseteq \mathcal{R}(t)$ instead.
}

\noindent
A full section is disproportionate for a short problem statement that is no longer than a single paragraph.
Only use a separate section if the statement requires additional definitions or detailed explanation.
